// Runtime: 1ms (Beats 100%)
// Memory: 45.15mb (Beats 8.76%)


class Solution {
    public List<Integer> findWordsContaining(String[] words, char x) {
        
        ArrayList<Integer> nIndex = new ArrayList<>();

        for (int i = 0; i < words.length; i++) {
            if (words[i].indexOf(x) != -1) {
                nIndex.add(i);
            }
        }
        return nIndex;
    }
    
}
